// ADC Project

#include "ADC.h"
#include "Bitwise.h"
#include "DIO.h"
#include "std_types.h"


int main(void)
{

	ADCSRA = 0b00000111;
	uint64 value = 0;

	Set_Bit(DDRD_REG, 0);
	Set_Bit(DDRD_REG, 1);
	Set_Bit(DDRD_REG, 3);
	Set_Bit(ADCSRA, ADEN);

	while (1)
	{
		ADMUX = 0b01000000;
		Set_Bit(ADCSRA, ADSC);

		while(Read_Bit(ADCSRA, ADSC));

		value = (ADCHL * 50) / 1024;

		if (value >= 0 && value <= 10)
		{
			Set_Bit(PORTD_REG,0);
			Clr_Bit(PORTD_REG,1);
			Clr_Bit(PORTD_REG,3);
		}
		else if (value > 10 && value <= 30)
		{
			Set_Bit(PORTD_REG,1);
			Clr_Bit(PORTD_REG,0);
			Clr_Bit(PORTD_REG,3);
		}
		else if(value > 30)
		{
			Set_Bit(PORTD_REG,3);
			Clr_Bit(PORTD_REG,1);
			Clr_Bit(PORTD_REG,0);
		}

	}
	return (0);
}
